<h2>Employee</h2>

<input type="hidden" id="empId" />

<input type="text" id="name" placeholder="Name" />
<input type="number" id="age" placeholder="Age" />
<input type="text" id="gender" placeholder="Gender" />
<input type="text" id="designation" placeholder="Designation" />

<button onclick="save()">Save</button>

<br />
<br />

<table border="1">
    <thead>
        <tr>
            <th>Name</th>
            <th>Age</th>
            <th>Gender</th>
            <th>Designation</th>
            <th>Action</th>
        </tr>
    </thead>
    <tbody id="empTable"></tbody>
</table>

<script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>

<script>
    $(document).ready(function () {
    loadData();
    });

    function loadData() {
    $.get('/EmployeesAjx/GetEmployees', function (data) {
    bindTable(data);
    });
    }

    function bindTable(data) {
    let rows = '';
    data.forEach(e => {
    rows += `<tr>
    <td>${e.name}</td>
    <td>${e.age ?? ''}</td>
    <td>${e.gender ?? ''}</td>
    <td>${e.designation ?? ''}</td>
    <td>
    <button onclick="edit(${e.id},'${e.name}','${e.age}','${e.gender}','${e.designation}')">Edit</button>
    <button onclick="deleteEmp(${e.id})">Delete</button>
    </td>
    </tr>`;
    });
    $('#empTable').html(rows);
    }

    function save() {
    let emp = {
    Id: $('#empId').val(),
    Name: $('#name').val(),
    Age: $('#age').val(),
    Gender: $('#gender').val(),
    Designation: $('#designation').val()
    };

    let url = emp.Id ? '/EmployeesAjx/Edit' : '/EmployeesAjx/Create';

    $.post(url, emp, function (data) {
    bindTable(data);
    clearForm();
    alert("Save Successful");
    });
    }

    function edit(id, name, age, gender, designation) {
    $('#empId').val(id);
    $('#name').val(name);
    $('#age').val(age);
    $('#gender').val(gender);
    $('#designation').val(designation);
    }

    function deleteEmp(id) {
        if(confirm("Are you sure want to delete this record."))
        {
            $.post('/EmployeesAjx/Delete', { id: id }, function (data) {
            bindTable(data);
            });
        }
    }

    function clearForm() {
    $('#empId').val('');
    $('#name').val('');
    $('#age').val('');
    $('#gender').val('');
    $('#designation').val('');
    }
</script>
